Z		Slash
X		Combo
Down 		Crouch
-> 		Move Right
<-		Move Left

-> or <- + down		Slide

Slide + C		Low Throw

-> or <- + Z		Moving Slash

2nd Moving Slash + X	Flurry of attacks
-> or <- + C		Moving Throw
Moving Slash + C	2nd Moving Throw
2nd Moving Slash + C	Triple Throw
